<script>
  var typed = new Typed('#typed', {
    strings: ['<strong>software engineer<strong>^1000','<strong>web/apps developer<strong>^1000','<strong>writer<strong>^1000'],
    typeSpeed: 40,
    loop: true,
    loopCount: Infinity,
  });

  // Toggle company details function
  function toggleCompany(companyElement) {
    const detailsDiv = companyElement.querySelector('.company-details');
    const toggleIcon = companyElement.querySelector('.toggle-icon i');
    const isExpanded = detailsDiv.style.display === 'block';
    
    if (isExpanded) {
      // Collapse
      detailsDiv.classList.add('collapsing');
      detailsDiv.classList.remove('expanding');
      
      setTimeout(() => {
        detailsDiv.style.display = 'none';
        detailsDiv.classList.remove('collapsing');
      }, 300);
      
      toggleIcon.className = 'fas fa-chevron-down';
      companyElement.querySelector('.toggle-icon').classList.remove('rotated');
    } else {
      // Expand
      detailsDiv.style.display = 'block';
      detailsDiv.classList.add('expanding');
      detailsDiv.classList.remove('collapsing');
      
      setTimeout(() => {
        detailsDiv.classList.remove('expanding');
      }, 300);
      
      toggleIcon.className = 'fas fa-chevron-up';
      companyElement.querySelector('.toggle-icon').classList.add('rotated');
    }
  }

  // Initialize all companies as collapsed on page load
  document.addEventListener('DOMContentLoaded', function() {
    const allCompanyDetails = document.querySelectorAll('.company-details');
    allCompanyDetails.forEach(details => {
      details.style.display = 'none';
    });

    // Expand Pepperstone (most recent company) by default
    const pepperstoneCompany = document.querySelector('#work-experience-pepperstone').closest('.collapsible-company');
    if (pepperstoneCompany) {
      const pepperstoneDetails = pepperstoneCompany.querySelector('.company-details');
      const pepperstoneIcon = pepperstoneCompany.querySelector('.toggle-icon i');
      const pepperstoneToggleIcon = pepperstoneCompany.querySelector('.toggle-icon');
      
      pepperstoneDetails.style.display = 'block';
      pepperstoneIcon.className = 'fas fa-chevron-up';
      pepperstoneToggleIcon.classList.add('rotated');
    }

    // Add click event listeners to all company containers
    const allCompanies = document.querySelectorAll('.collapsible-company');
    allCompanies.forEach(company => {
      company.addEventListener('click', function() {
        toggleCompany(this);
      });
    });
  });
</script>
